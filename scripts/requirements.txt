#!/usr/bin/env python3
"""
Requirements installer for bug hunter tools
Usage: python install.py
"""

import subprocess
import sys
import os

def install_packages():
    """Install required Python packages"""
    packages = [
        'requests',
        'urllib3',
        'dnspython',
        'colorama',
    ]
    
    # Optional packages for extended functionality
    optional = [
        'dnspython',  # DNS lookups
        'pwned-or-not',  # Check password breaches
        'shodan',  # Shodan API
    ]
    
    print("[*] Installing required packages...")
    
    for package in packages:
        try:
            subprocess.check_call([sys.executable, '-m', 'pip', 'install', package])
            print(f"[+] Installed: {package}")
        except:
            print(f"[!] Failed to install: {package}")
    
    print("\n[*] Installing optional packages (y/n)?")
    ans = input("> ")
    
    if ans.lower() == 'y':
        for package in optional:
            try:
                subprocess.check_call([sys.executable, '-m', 'pip', 'install', package])
                print(f"[+] Installed: {package}")
            except:
                print(f"[!] Failed to install: {package}")

def check_tools():
    """Check if Go tools are available"""
    tools = [
        'subfinder',
        'httpx',
        'nuclei',
        'katana',
        'ffuf',
    ]
    
    print("\n[*] Checking Go tools...")
    
    for tool in tools:
        try:
            result = subprocess.run([tool, '-h'], capture_output=True)
            if result.returncode == 0:
                print(f"[+] Found: {tool}")
            else:
                print(f"[-] Not found: {tool}")
        except FileNotFoundError:
            print(f"[-] Not found: {tool}")


def main():
    print("="*50)
    print("Bug Hunter - Requirements Installer")
    print("="*50)
    
    install_packages()
    check_tools()
    
    print("\n[+] Setup complete!")
    print("\n[*] To use the tools:")
    print("    python tools/python/*.py <target>")
    print("    powershell tools/bug-hunter/*.ps1 -Target <target>")


if __name__ == "__main__":
    main()
